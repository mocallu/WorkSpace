# Transparência
``` bash
$((0xffffffff * 99 / 100)))

sudo apt-get install devilspie
mkdir -p ~/.devilspie
nano ~/.devilspie/transparent.ds

( if
( contains ( window_class ) "Code" )
( begin
( spawn_async (str "xprop -id " (window_xid) " -f _NET_WM_WINDOW_OPACITY 32c -set _NET_WM_WINDOW_OPACITY 0xf3333332") )
)
)
```
